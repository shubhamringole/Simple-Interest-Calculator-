# Simple Intrest Calculator

P = float(input('Principal(P) $ \n'))
R = float(input('Rate (R):% \n(per year) \n'))
T = float(input('Time (t)'))

A = P*(1 + (R * T))

print(f'The total amount accrued, principal plus interest, from simple interest on a principal of $ {P} at a rate of {R}% per year for {T} years is ${A}')
